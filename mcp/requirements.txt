# =============================================================================
# mcp/requirements.txt - MCPツールサーバーの依存ライブラリ
# =============================================================================

# 【FastAPI】WebAPIフレームワーク
# 【用途】APIサーバーからMCPツールを呼び出すためのHTTPエンドポイント
fastapi>=0.115.0

# 【uvicorn】FastAPIを動かすためのASGIサーバー
uvicorn>=0.34.0

# 【FastMCP】MCPサーバーを簡単に作るためのフレームワーク
# 【バージョン指定 <3】
#   FastMCP 3.0は破壊的変更がある可能性があるため、2.x系を使用
#   参考: https://github.com/jlowin/fastmcp
fastmcp>=2.0.0,<3
# 【fakeredis バージョン固定】
# fastmcp 2.14.5 が依存する pydocket が fakeredis を使うが、
# fakeredis 2.34.x で FakeConnection が削除されて起動エラーになる。
# 互換性のあるバージョンに固定する。
fakeredis>=2.21.0,<2.26.0

# 【plotly】インタラクティブグラフ描画ライブラリ
# 【用途】
#   - stripplot風の散布図を生成し、HTMLファイルとして保存
#   - ホバー（カーソルを合わせると値が表示される）機能が標準搭載
# 【なぜplotlyか】
#   - HTML出力 → フロントエンド非依存（MCP側がStreamlitを知らなくて良い）
#   - ホバーでx,y値と凡例が表示される（seaborn/matplotlibのPNGでは不可能）
#   - 将来フロントが変わっても、HTMLファイルをそのまま使える
# 【他の選択肢】
#   - matplotlib/seaborn: PNG画像のみ。インタラクティブ機能なし
#   - bokeh: plotlyと同様にインタラクティブだが、エコシステムが小さい
plotly>=5.18.0

# 【pandas】データ分析ライブラリ
# 【用途】
#   - plotly.expressに渡すDataFrame（表形式データ）の作成
#   - plotly.expressはpandasのDataFrameを入力として期待する
pandas>=2.0.0

# 【prophet】Meta（Facebook）製の時系列予測ライブラリ
# 【用途】
#   - 点検データの測定値トレンドを統計モデルで予測する
#   - forecast_time_series MCPツールで使用
# 【なぜProphetか】
#   - 数式ベースなので毎回同じ結果（再現性がある）
#   - トレンド + 季節性を自動分離できる
#   - 変化点（トレンドが途中で変わる）も検出可能
# 【他の選択肢】
#   - LLM予測: 手軽だが毎回結果が変わる（既存実装として併用）
#   - scikit-learn: 汎用的だが時系列特化ではない
#   - statsmodels(ARIMA): 古典的だが設定が複雑
prophet>=1.1.5

# 【httpx】非同期HTTPクライアント
# 【用途】
#   - test_iframe MCPツールからWatson Assistant APIを呼び出す
# 【なぜhttpxか】
#   - async/await対応（FastMCPのasyncツール内で使える）
#   - requestsはasync非対応
httpx>=0.25.0

# 【scipy】科学計算ライブラリ
# 【用途】
#   - curve_fit（カーブフィッティング）で測定値の劣化カーブを推定する
#   - forecast_curve_fit MCPツールで使用
# 【なぜscipyか】
#   - 線形だけでなく指数・対数カーブにもフィットできる
#   - 少ないデータ点でも動作する（Prophetは大量データ向き）
#   - 改修サイクルごとに正規化したデータに対してカーブを当てはめる用途に最適
# 【他の選択肢】
#   - numpy.polyfit: 多項式のみ。指数・対数カーブには対応できない
#   - scikit-learn: 汎用的だが、カーブフィッティング特化ではない
scipy>=1.11.0
